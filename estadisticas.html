// Generar Datos para GrÃ¡fica de Cursos (Enero P1/P2, Junio P1/P2, Agosto P1/P2, Total) const generateCourseEvolutionData = () => { if (!stats || !stats.historicalTrend) return {}; // Filtrar para mostrar desde 2022 a 2025 preferentemente si existen const startYear = 2022; const years = Object.keys(stats.historicalTrend).sort().filter(y => parseInt(y) >= startYear && parseInt(y) <= 2025); let labels = []; let dataPoints = []; let backgroundColors = []; // Paleta de Colores const colorEne = '#FF6384'; // Rosa/Rojo suave const colorJun = '#36A2EB'; // Azul const colorAgo = '#FFCE56'; // Amarillo const colorTotal = '#1B396A'; // Azul Institucional (Total) years.forEach(year => { const totalCursos = stats.historicalTrend[year].cursos || 0; // Extraer datos de periodos (ahora por mes directo) const pEnero = stats.periodTrend?.[`${year}-Enero`]?.cursos || 0; const pJunio = stats.periodTrend?.[`${year}-Junio`]?.cursos || 0; const pAgosto = stats.periodTrend?.[`${year}-Agosto`]?.cursos || 0; // Orden solicitado: Total 'YY, Ene, Jun, Ago // Total Annual labels.push(`Total '${year.substring(2)}`); dataPoints.push(totalCursos); backgroundColors.push(colorTotal); // Enero labels.push(`Ene`); dataPoints.push(pEnero); backgroundColors.push(colorEne); // Junio labels.push(`Jun`); dataPoints.push(pJunio); backgroundColors.push(colorJun); // Agosto labels.push(`Ago`); dataPoints.push(pAgosto); backgroundColors.push(colorAgo); }); return { labels: labels, datasets: [ { label: 'Cursos Ofertados', data: dataPoints, backgroundColor: backgroundColors, borderRadius: 4, barPercentage: 0.9, categoryPercentage: 0.95 } ] }; };
